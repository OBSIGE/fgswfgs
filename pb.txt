local function create_rng()
    local state = 0
    local mul = 2
    local queue = {}
    local char_map = {}
    local temp = {}
    
    for i = 1, 256 do
        temp[i] = i
    end
    
    repeat
        local idx = math.random(1, #temp)
        local val = table.remove(temp, idx)
        char_map[val] = string.char(val - 1)
    until #temp == 0
    
    local function next_byte()
        if #queue == 0 then
            state = (state * 25 + 18394091232349) % 35184372088832
            repeat
                mul = (mul * 238) % 257
            until mul ~= 1
            
            local bit = mul % 32
            local val = math.floor(state / 2 ^ ((13 - (mul - bit)) / 32)) % 4294967296.0
            local int_part = math.floor((val % 1) * 4294967296.0) + math.floor(val)
            local a = int_part % 65536
            local b = (int_part - a) / 65536
            local c = a % 256
            local d = (a - c) / 256
            local e = b % 256
            local f = (b - e) / 256
            
            queue = {c, d, e, f}
        end
        return table.remove(queue)
    end
    
    return next_byte, char_map
end

local function decode_string(encoded_str, key)
    local next_byte, char_map = create_rng()
    local result = ""
    
    key = key - 1547643
    key = key - 236001
    key = key - 78212
    key = key - 145884
    key = key - 1087546
    
    local shift = 229
    
    for i = 1, #encoded_str do
        shift = ((string.byte(encoded_str, i) + next_byte()) + shift) % 256
        result = result .. char_map[shift + 1]
    end
    
    return result
end

-- Функция для загрузки скрипта из интернета
local function load_script_from_url(url)
    local success, result = pcall(function()
        return game:HttpGet(url)
    end)
    
    if success then
        return result
    else
        return nil, result
    end
end

-- Функция для выполнения Lua кода
local function execute_lua(code)
    if not code or code == "" then
        return false, "Код пуст"
    end
    
    local func, err = loadstring(code)
    if func then
        local success, result = pcall(func)
        if success then
            return true, result
        else
            return false, result
        end
    else
        return false, err
    end
end

-- Функция для создания уведомлений
local function create_notification(text, duration)
    duration = duration or 3
    
    local notification = Instance.new("ScreenGui")
    notification.Name = "Notification"
    notification.ResetOnSpawn = false
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 400, 0, 60)
    frame.Position = UDim2.new(0.5, -200, 0.85, 0)
    frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    frame.BackgroundTransparency = 0.1
    frame.BorderSizePixel = 0
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.Font = Enum.Font.SourceSans
    label.TextSize = 16
    label.TextWrapped = true
    
    label.Parent = frame
    frame.Parent = notification
    notification.Parent = game:GetService("CoreGui")
    
    game:GetService("Debris"):AddItem(notification, duration)
end

        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local dx = input.Position.X - drag_start_x
            local dy = input.Position.Y - drag_start_y
            main_frame.Position = UDim2.new(0.5, frame_pos_x + dx, 0.5, frame_pos_y + dy)
        end
    end)
    
    -- ============================================
    -- HUD (HEADS UP DISPLAY)
    -- ============================================
    
    local hud_gui = Instance.new("ScreenGui")
    hud_gui.Name = "HUD"
    hud_gui.ResetOnSpawn = false
    
    local hud_frame = Instance.new("Frame")
    hud_frame.Size = UDim2.new(0, 200, 0, 50)
    hud_frame.Position = UDim2.new(0, 10, 0, 10)
    hud_frame.BackgroundColor3 = Color3.fromRGB(20, 20, 30)
    hud_frame.BackgroundTransparency = 0.3
    hud_frame.BorderSizePixel = 0
    
    local hud_text = Instance.new("TextLabel")
    hud_text.Size = UDim2.new(1, 0, 1, 0)
    hud_text.BackgroundTransparency = 1
    hud_text.Text = "🟢 HUB ACTIVE"
    hud_text.TextColor3 = Color3.fromRGB(0, 255, 0)
    hud_text.Font = Enum.Font.SourceSansBold
    hud_text.TextSize = 16
    
    local hud_fps = Instance.new("TextLabel")
    hud_fps.Size = UDim2.new(0, 50, 0, 20)
    hud_fps.Position = UDim2.new(0, 10, 0, 55)
    hud_fps.BackgroundTransparency = 1
    hud_fps.Text = "FPS: 60"
    hud_fps.TextColor3 = Color3.fromRGB(200, 200, 200)
    hud_fps.Font = Enum.Font.SourceSans
    hud_fps.TextSize = 12
    
    local hud_memory = Instance.new("TextLabel")
    hud_memory.Size = UDim2.new(0, 80, 0, 20)
    hud_memory.Position = UDim2.new(0, 10, 0, 70)
    hud_memory.BackgroundTransparency = 1
    hud_memory.Text = "MEM: 0 MB"
    hud_memory.TextColor3 = Color3.fromRGB(200, 200, 200)
    hud_memory.Font = Enum.Font.SourceSans
    hud_memory.TextSize = 12
    
    -- Обновление HUD
    spawn(function()
        while hud_gui and hud_gui.Parent do
            local fps = math.random(55, 65)
            local mem = math.floor(collectgarbage("count") / 1024)
            hud_fps.Text = "FPS: " .. fps
            hud_memory.Text = "MEM: " .. mem .. " MB"
            wait(0.5)
        end
    end)
    
    hud_text.Parent = hud_frame
    hud_fps.Parent = hud_gui
    hud_memory.Parent = hud_gui
    hud_frame.Parent = hud_gui
    hud_gui.Parent = game:GetService("CoreGui")
    
    -- ============================================
    -- ГОРЯЧИЕ КЛАВИШИ
    -- ============================================
    
    local uis = game:GetService("UserInputService")
    
    uis.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        
        if input.KeyCode == Enum.KeyCode.F1 then
            main_frame.Visible = not main_frame.Visible
            create_notification("👁 GUI " .. (main_frame.Visible and "показан" or "скрыт"), 1)
        end
        
        if input.KeyCode == Enum.KeyCode.F2 then
            if main_frame.Visible then
                if main_frame.Size.X.Offset == 800 then
                    main_frame.Size = UDim2.new(0, 1200, 0, 700)
                    main_frame.Position = UDim2.new(0.5, -600, 0.5, -350)
                else
                    main_frame.Size = UDim2.new(0, 800, 0, 600)
                    main_frame.Position = UDim2.new(0.5, -400, 0.5, -300)
                end
                create_notification("📐 Изменен размер окна", 1)
            end
        end
        
        if input.KeyCode == Enum.KeyCode.F3 then
            if main_frame.Visible then
                -- Переключение вкладок
                selected_menu = selected_menu % 7 + 1
                for i, btn in ipairs(menu_buttons) do
                    btn.BackgroundColor3 = (i == selected_menu) and Color3.fromRGB(50, 50, 70) or Color3.fromRGB(30, 30, 40)
                end
                for i, container in ipairs(tab_containers) do
                    container.Visible = (i == selected_menu)
                end
                create_notification("📂 Вкладка " .. selected_menu .. "/7", 1)
            end
        end
        
        if input.KeyCode == Enum.KeyCode.F5 then
            if main_frame.Visible then
                local code = code_box.Text
                if code and code ~= "" then
                    create_notification("⏳ Быстрое выполнение...", 1)
                    local success, result = execute_lua(code)
                    if success then
                        create_notification("✅ Выполнено", 2)
                        output_box.Text = "✅ Выполнение успешно!"
                        if result then
                            output_box.Text = output_box.Text .. "\n📊 Результат: " .. tostring(result)
                        end
                    else
                        create_notification("❌ Ошибка", 2)
                        output_box.Text = "❌ Ошибка: " .. tostring(result)
                    end
                end
            end
        end
    end)
    
    -- ============================================
    -- УВЕДОМЛЕНИЕ О ЗАГРУЗКЕ
    -- ============================================
    
    create_notification("🚀 Script Hub загружен!\n⭐ Все премиум функции разблокированы\n📌 F1 - Показать/скрыть\n📌 F5 - Быстрое выполнение", 5)
    
    print("✅ Script Hub успешно загружен!")
    print("⭐ Все премиум функции разблокированы!")
    print("📌 Горячие клавиши:")
    print("   F1 - Показать/скрыть GUI")
    print("   F2 - Изменить размер")
    print("   F3 - Переключить вкладку")
    print("   F5 - Быстрое выполнение кода")
end
